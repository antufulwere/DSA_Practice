#include <iostream>
using namespace std;

void fun(int n){
    //base condition
    if(n==0) {
        return;
    } 
    // expectation + feith
    cout<<n<<" ";   // print 5 is expectation
    fun(n-1);   // print 4,3,2,1 is feith
}

int main() {

    int n=5;
    fun(n);

}
