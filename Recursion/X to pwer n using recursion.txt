// Online C++ compiler to run C++ program online
// below code complexity is O(n)
#include <iostream>
using namespace std;

int fun(int x, int n){
    // base condition 
    if(n==0){
        return 1;
    }
    //exeprcttion + feith
    return x * fun(x,n-1);
}
int main() {
    int x=2;
    int n=0;
    cout<<fun(x,n);
    return 0;
}

// Online C++ compiler to run C++ program online
#include <iostream>
using namespace std;

int fun(int x, int n){
    // base condition 
    if(n==0){
        return 1;
    }
    //exeprcttion + feith
    int xpn=fun(x,n/2)*fun(x,n/2);
    if(n%2==1){
        xpn=xpn*x;
    }
    return xpn;
}
int main() {
    int x=2;
    int n=3;
    cout<<fun(x,n);
    return 0;
}
